var getNumber = parseInt(prompt("Give a number"));

        var ul = "<ul>";
        if(!isNaN(getNumber)){
            for( i =1; i <= 10; i++){
                //console.log(getNumber + "*" + i + "=" + (i * getNumber));
                var res = "<li>" + getNumber + "*" + i + "=" + (i * getNumber) + "</li>";
                ul += res;    
            }
        ul += "</ul>";
        document.write(ul);
        }else{
            alert("Number only!!!!");
        }
