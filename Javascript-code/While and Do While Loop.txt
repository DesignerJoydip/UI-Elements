var i = 5;

while(i <= 10){
 console.log(i);
 i++;
}
// result: 1, 2, 3, 4 ...10


var i = 15;

while (i <= 10) {
 console.log(i);
 i++;
}
// result will not execute.

-----------------------------------------------------------------
Do While
-----------------------------------------------------------------
var i = 15;

do{
 console.log(i);
 i++;
}while(i <= 10);
// result: 15


var i = 5;

do{
 console.log(i);
 i++;
}while(i <= 10);
// result: 5, 6 .... 10
