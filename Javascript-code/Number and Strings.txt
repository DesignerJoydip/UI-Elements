var age = prompt("Your Age");
console.log("After 5 year, you will be: " + (age + 5) );
// we gave value in prompt: 23.
Result = 235 * this is wrong
// because javascript get the number from prompt as a string.


Solution:
We use parseInt();

var age;
age = prompt("Your Age");
console.log(typeof age); // for check the data type
age = parseInt(age);
console.log(typeof age); // for check the data type
console.log("After 5 year, you will be: " + (age + 5) );
// we gave value in prompt: 23.
Result: 28
